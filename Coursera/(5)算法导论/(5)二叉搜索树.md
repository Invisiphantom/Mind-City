

# 二叉搜索树

```py
import os
from graphviz import Digraph
from IPython.display import Image, display


class Node:
    def __init__(self, key, data):
        self.key = key
        self.data = data
        self.p = None
        self.left = None
        self.right = None


class BST:
    def __init__(self):
        self.root = None

    def insert(self, key, data):
        """插入带有键的结点"""

        # 搜索直到匹配或空结点
        pNode = None
        node = self.root
        while node and key != node.key:
            pNode = node
            if key < node.key:
                node = node.left
            else:
                node = node.right

        # 如果键已存在, 则更新数据
        if node and key == node.key:
            node.data = data
            return

        # 建立当前结点的向上连接
        iNode = Node(key, data)
        iNode.p = pNode

        # 建立父结点的向下连接
        if pNode is None:
            self.root = iNode
        elif key < pNode.key:
            pNode.left = iNode
        elif key > pNode.key:
            pNode.right = iNode


    def search(self, key):
        """搜索键对应的结点"""
        node = self.root
        while node and node.key != key:
            if key < node.key:
                node = node.left
            else:
                node = node.right
        return node

    def _minimum(self, node):
        """返回子树的最小结点"""
        while node.left:
            node = node.left
        return node

    def _maximum(self, node):
        """返回子树的最大结点"""
        while node.right:
            node = node.right
        return node

    def successor(self, node):
        """返回后继结点"""
        # 直接返回右子树的最小结点
        if node.right:
            return self._minimum(node.right)

        # 向上搜索 直到首次右拐
        # 即首个稍大的父结点
        pNode = node.p
        while pNode and node == pNode.right:
            node = pNode
            pNode = pNode.p
        return pNode

    def predecessor(self, node):
        """返回前驱结点"""
        # 直接返回左子树的最大结点
        if node.left:
            return self._maximum(node.left)

        # 向上搜索 直到首次左拐
        # 即首个稍小的父结点
        pNode = node.p
        while pNode and node == pNode.left:
            node = pNode
            pNode = pNode.p
        return pNode

    def _transplant(self, old, new):
        """替换子树"""
        # 更新当前结点的向上连接
        if new:
            new.p = old.p

        # 更新父结点的向下连接
        if old.p is None:
            self.root = new
        elif old == old.p.left:
            old.p.left = new
        else:
            old.p.right = new

    def delete(self, node):
        """删除结点"""
        # 如果左子树为空, 则直接用右子树替换
        if node.left is None:
            self._transplant(node, node.right)
        # 如果右子树为空, 则直接用左子树替换
        elif node.right is None:
            self._transplant(node, node.left)

        # 如果左右子树都不为空, 则用后继结点替换
        else:
            # 如果后继不是直接子结点
            # 则需要移交后继结点的右子树
            nxt = self._minimum(node.right)
            if nxt.p != node:
                self._transplant(nxt, nxt.right)
                nxt.right = node.right
                nxt.right.p = nxt
            self._transplant(node, nxt)
            nxt.left = node.left
            nxt.left.p = nxt

    def inorder(self, node):
        """中序遍历"""
        if node:
            self.inorder(node.left)
            print(node.key)
            self.inorder(node.right)

    def draw_tree(self, dot=None, node=None):
        """绘制二叉树"""
        if dot is None:
            # 初始化图
            dot = Digraph()
            dot.attr("node", shape="circle")
            dot.attr("edge", dir="none")

            # 递归添加每层的结点和边
            self.draw_tree(dot, self.root)

            # 渲染并显示图像
            dot.render("bst", format="png", cleanup=True)
            display(Image("bst.png"))
            os.remove("bst.png")
        else:
            if node:
                str_node = str(node.key)
                dot.node(str_node, str_node + f"-{node.data}")
                if node.left:
                    str_left = str(node.left.key)
                    dot.edge(str_node, str_left, constraint="true")
                    self.draw_tree(dot, node.left)
                if node.right:
                    str_right = str(node.right.key)
                    dot.edge(str_node, str_right, constraint="true")
                    self.draw_tree(dot, node.right)
```
